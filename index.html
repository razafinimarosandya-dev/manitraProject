/** * GOOGLE APPS SCRIPT - BACKEND FINANCIER * À copier dans : https://script.google.com/ */ const FILE_IDS = { facturation: "ID_DU_FICHIER_FACTURATION", // Remplacez par l'ID réel achats: "ID_DU_FICHIER_ACHATS", // Remplacez par l'ID réel tresorerie: "ID_DU_FICHIER_TRESORERIE" // Remplacez par l'ID réel }; function doGet(e) { try { const action = e.parameter.action || "getFinancialSummary"; const callback = e.parameter.callback; let data; if (action === "getFinancialSummary") { data = getCombinedData(); } const output = JSON.stringify(data); // JSONP Support for the dashboard if (callback) { return ContentService.createTextOutput(callback + "(" + output + ")") .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService.createTextOutput(output) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService.createTextOutput(JSON.stringify({ error: err.toString() })) .setMimeType(ContentService.MimeType.JSON); } } function getCombinedData() { return { tresorerie: getTresorerieData(), achats: getAchatsData(), facturation: getFacturationData(), timestamp: new Date().toISOString() }; } /** * Analyse du fichier SUIVI_DE_TRESORERIE */ function getTresorerieData() { const ss = SpreadsheetApp.openById(FILE_IDS.tresorerie); const sheet = ss.getSheetByName("suivi trésorerie"); const data = sheet.getDataRange().getValues(); let currentBalance = 0; let evolution = []; // On cherche la ligne "SOLDE COMPTE COURANT" for (let i = 0; i < data.length; i++) { const rowLabel = String(data[i][0]).toUpperCase(); if (rowLabel.indexOf("SOLDE COMPTE COURANT") !== -1 || rowLabel.indexOf("SOLDE") !== -1) { currentBalance = data[i][15] || 0; // Colonne P (TOTAL) evolution = data[i].slice(2, 14); // Janvier (C) à Décembre (N) break; } } return { soldeActuel: currentBalance, evolutionMensuelle: evolution }; } /** * Analyse du fichier REGISTRE_DES_ACHATS */ function getAchatsData() { const ss = SpreadsheetApp.openById(FILE_IDS.achats); const sheet = ss.getSheetByName("RECAPITULATIF"); // On prend une plage large et on filtre les lignes vides const values = sheet.getRange("A3:D25").getValues(); const stats = { totalDepenses: 0, depensesMensuelles: [] }; values.forEach(row => { if (row[0] && row[0].toString().toUpperCase() !== "TOTAL") { const depense = parseFloat(row[2]) || 0; stats.depensesMensuelles.push(depense); stats.totalDepenses += depense; } }); return stats; } /** * Analyse du fichier FACTURATION_BBT */ function getFacturationData() { try { const ss = SpreadsheetApp.openById(FILE_IDS.facturation); const sheet = ss.getSheetByName("Chiffre affaires"); const data = sheet.getDataRange().getValues(); // Extraction simplifiée du premier montant significatif (Total CA) // Selon l'analyse, Row 0 Col 0 semble être 1 781 000 let totalCA = data[0][0] || 0; return { totalCA: totalCA, enAttente: totalCA * 0.12 // Estimation ou calcul à affiner selon Suivi des paiements }; } catch (e) { return { totalCA: 0, enAttente: 0, error: e.toString() }; } }
